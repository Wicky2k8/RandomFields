\def\s#1{

  {\loop
    \edef\nra{\the\nr} \advance \nr by 1
    \edef\nrb{\the\nr} \advance \nr by 1
                                %   \edef\nrc{\the\nr} \advance \nr by 1
    \x \nra \nrb % \nrc 
    \ifnum\nr<\the\total
    \repeat}
  }
